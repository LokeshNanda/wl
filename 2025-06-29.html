  llm_transformer = LLMGraphTransformer(llm=chat)
  graph_documents = llm_transformer.convert_to_graph_documents(documents)

  # store to neo4j
  res = kg.add_graph_documents(
    graph_documents,
    include_source=True,
    baseEntityLabel=True,
